  (defun my/truncate-string (s &optional len)
    "Return at most LEN chars of S, appending “…” if truncated."
    (let ((len (or len 300)))
      (if (> (length s) len)
	  (concat (substring s 0 len) "…")
	s)))

  (defun my/org-roam-search (query &optional begin-range end-range)
    "Search org-roam nodes matching QUERY with pagination.
  BEGIN-RANGE and END-RANGE are one-based indices (inclusive).
  Defaults to 1–10."
    (require 'seq)
    (let* ((nodes  (org-roam-node-list))
	   (matches
	    (seq-filter
	     (lambda (node)
	       (or (string-match-p query (org-roam-node-title node))
		   (string-match-p query
				   (or (string-join (org-roam-node-tags node) " ")
				       ""))
		   (with-temp-buffer
		     (insert-file-contents (org-roam-node-file node))
		     (string-match-p query (buffer-string)))))
	     nodes))
	   (total  (length matches))
	   (start  (or begin-range 0))
	   (end    (or end-range 10)))
      (if (zerop total)
	  "No matching nodes found."
	(let* ((end-excl (min total (1+ end)))
	       (paged    (seq-subseq matches (min start total) end-excl))
	       (header   (format "Showing results %d–%d of %d matches:\n\n"
				 (1+ start)
				 (min end total)
				 total)))
	  (concat header
		  (mapconcat
		   (lambda (node)
		     (format "Title: %s\nTags: %s\nFile: %s\nSnippet:\n%s\n---\n"
			     (org-roam-node-title node)
			     (string-join (or (org-roam-node-tags node) '("None")) ", ")
			     (org-roam-node-file node)
			     (with-temp-buffer
			       (insert-file-contents (org-roam-node-file node))
			       (my/truncate-string (buffer-string) 300))))
		   paged
		   "\n"))))))

(require 'cl-lib)

(cl-defun my/create-memory (&key title memory-type tags aliases content)
  "Create an org-roam node with validated metadata."
  (let ((valid-types
         '(episodic semantic procedural associative working reflective reference)))
    ;; 1) TITLE: non-empty string
    (unless (and (stringp title) (> (length (string-trim title)) 0))
      (error "create-memory error: TITLE must be a non-empty string"))
    ;; 2) MEMORY-TYPE: must be one of valid-types
    (unless (memq memory-type valid-types)
      (error "create-memory error: MEMORY-TYPE must be one of %S" valid-types))
    ;; 3) CONTENT: must be a string
    (unless (stringp content)
      (error "create-memory error: CONTENT must be a string"))

    (require 'org-id)
    ;; All bindings in one let* so filepath is in scope for the org-roam calls:
    (let* ((slug     (downcase
                      (replace-regexp-in-string
                       "[^a-z0-9]+" "-"
                       (string-trim title))))
           (id (org-id-new))
           (filename (format "%s.%s.org" (format-time-string "%Y-%m-%d-%H-%M-%S") slug))
           (filepath (expand-file-name filename org-roam-directory))
           (lowercased-tags (mapcar #'downcase tags))
           (filetags (mapconcat
                       #'identity
                       (cons (symbol-name memory-type) lowercased-tags)
                       " "))
           (props    `(("ID"            . ,id)

                       ("ROAM_ALIASES"  . ,(when aliases
                                             (concat "\""
                                                     (mapconcat #'identity aliases "\" \"")
                                                     "\"")))
                       ("CREATED"       . ,(format-time-string "<%Y-%m-%d %H:%M>"))
                       ("LAST_MODIFIED" . ,(format-time-string "<%Y-%m-%d %H:%M>")))))
      ;; 1) Create the node object (does not write file yet)
      (org-roam-node-create
       :id         id
       :title      title
       :file       filepath
       :aliases    aliases
       :properties props)
      ;; 2) Write out the file with front-matter + content
      (with-temp-file filepath
        (insert ":PROPERTIES:\n")
        (dolist (prop props)
          (insert (format ":%s: %s\n" (car prop) (cdr prop))))
        (insert ":END:\n")
        (insert (format "#+TITLE: %s\n" title))
        (insert (format "#+FILETAGS: %s\n" filetags))
        (insert "\n" content "\n"))
      ;; 3) Sync to the database
      (org-roam-db-sync)
      ;; Return metadata for callers and also message
      (let ((ret (list :id id :file filepath :title title)))
        (message "create-memory: Node \"%s\" created at %s. Node ID: %s"
                 title filepath id)
        ret))))

      ;; Show the created memory info to user

               (plist-get memory-result :title)
               (plist-get memory-result :id))))

(defun my/create-conversation-from-heading ()
  "Create a new conversation from a selected heading and link it in Conversation History.
This function automates the workflow of:
1. Selecting a heading path (like org-copy-heading-path)
2. Creating a conversation memory with that heading
3. Adding a link to the conversation in the current buffer's 'Conversation History' section

Requires a 'Conversation History' section in the current buffer, otherwise errors."
  (interactive)
  (unless (derived-mode-p 'org-mode)
    (user-error "Not in an org-mode buffer"))
  
  (let ((original-buffer (current-buffer))
        headings)
    
    ;; Collect all headings with their full paths (reusing org-copy-heading-path logic)
    (org-map-entries
     (lambda ()
       (let* ((heading (org-get-heading t t t t)) ; Clean heading without TODO, tags, etc.
              (path (org-get-outline-path))        ; Get parent path
              (full-path (append path (list heading))) ; Combine parents + current heading
              (formatted-path (mapconcat 'identity (reverse full-path) " > ")) ; Reverse for ascending order
              (pos (point)))
         (push (cons formatted-path pos) headings))))
    
    (if (not headings)
        (message "No headings found in buffer")
      
      ;; Let user select a heading  
      (let* ((choice (completing-read "Select heading for conversation: " 
                                     (mapcar #'car (nreverse headings)) 
                                     nil t))
             (selected-path choice)
             (title (concat "Conversation: " selected-path)))
        
        ;; Create the memory node (replicating my/create-named-gptel logic)
        (let ((memory-result 
               (my/create-memory
                :title title
                :memory-type 'episodic
                :tags '("conversation")
                :aliases '()
                :content "")))
          
          ;; Open the memory file and enable gptel-mode
          (find-file (plist-get memory-result :file))
          (gptel-mode 1)
          (goto-char (point-max))
          
          ;; Add link to original buffer's Conversation History section
          (with-current-buffer original-buffer
            (save-excursion
              (goto-char (point-min))
              (unless (re-search-forward "^\\*+ Conversation History$" nil t)
                (user-error "No 'Conversation History' section found in current buffer"))
              
              ;; Use org-end-of-subtree to find the end of this section
              (let ((subtree-end (save-excursion (org-end-of-subtree t nil))))
                (goto-char subtree-end)
                ;; Skip backwards over any trailing whitespace
                (skip-chars-backward " \t\n\r")
                (end-of-line)
                ;; Insert our new list item
                (insert (format "\n- [[id:%s][%s]]" 
                              (plist-get memory-result :id)
                              selected-path)))))
          
          (message "Created conversation and added to Conversation History: %s" title))))))

(defun my/gptel-get-single-memory-content (id)
  "Get just the content of a single memory without processing required reading."
  (let ((node (org-roam-node-from-id id)))
    (when node
      (let* ((file (org-roam-node-file node))
             (title (org-roam-node-title node)))
        (with-temp-buffer
          (insert-file-contents file)
          (org-mode)
          (goto-char (org-roam-node-point node))
          (let ((content (buffer-substring-no-properties
                          (point)
                          (progn (org-end-of-subtree t) (point)))))
            ;; Post-process to escape CLOCK lines to prevent org-clock confusion
            (setq content (replace-regexp-in-string "^\\(CLOCK:\\)" ",\\1" content))
            (list :id id :title title :content content :file file)))))))

(defun my/gptel-extract-required-reading-links (content file)
  "Extract required reading links from memory content."
  (let (required-reading)
    (with-temp-buffer
      (insert content)
      (org-mode)
      (goto-char (point-min))
      (let ((case-fold-search t))
        (when (re-search-forward "^\\(\\*+\\) Required Reading\\s-*$" nil t)
          (let* ((level (length (match-string 1)))
                 (start (point))
                 (end-pattern (format "^\\*\\{1,%d\\} " level))
                 (end (or (save-excursion 
                            (re-search-forward end-pattern nil t))
                          (point-max))))
            (let ((section (buffer-substring-no-properties start end)))
              (with-temp-buffer
                (insert section)
                (goto-char (point-min))
                (while (re-search-forward
                        "- +\\[\\[\\(id:[^]]+\\|file\\(?:\\+[^:]+\\)?:[^]]+\\|/[^]]+\\|jira:[^]]+\\)\\]\\(?:\\[\\([^]]*\\)\\]\\)?\\]"
                        nil t)
                  (let ((link (match-string 1))
                        (desc (or (match-string 2) "(No description)")))
                    (push (cons link desc) required-reading)))))))))
    (reverse required-reading)))

(defun my/gptel-process-required-reading-links (links seen file-seen base-file)
  "Process required reading links and return a list of title-keyed memory data."
  (let (results)
    (dolist (link-pair links)
      (let* ((raw (car link-pair))
             (desc (cdr link-pair)))
        (cond
         ;; Org-roam node: follow id:
         ((and (string-prefix-p "id:" raw)
               (not (member raw seen)))
          (let* ((real-id (substring raw 3))
                 (child-result (my/gptel-get-single-memory-content real-id)))
            (when child-result
              (let* ((child-title (plist-get child-result :title))
                     (child-links (my/gptel-extract-required-reading-links 
                                   (plist-get child-result :content)
                                   (plist-get child-result :file)))
                     (nested-results (my/gptel-process-required-reading-links 
                                      child-links (cons raw seen) file-seen base-file)))
                ;; Add the child memory with title as key
                (push (list (intern child-title) child-result) results)
                ;; Add any nested required reading
                (setq results (append results nested-results))))))
         ;; Jira ticket pattern: jira:PM-N format
         ((string-prefix-p "jira:" raw)
          (let* ((ticket-key (substring raw 5)))
            (condition-case err
                (let ((jira-content (shell-command-to-string
                                   (format "jira issue view %s --raw" ticket-key))))
                  (when (and jira-content (> (length jira-content) 0))
                    (let ((jira-data (list :id ticket-key
                                          :title (format "Jira Ticket: %s (Raw JSON)" ticket-key) 
                                          :content jira-content
                                          :file (format "jira://%s" ticket-key))))
                      (push (list (intern (format "Jira-%s" ticket-key)) jira-data) results))))
              (error
               (message "Failed to fetch Jira ticket %s: %s" ticket-key (error-message-string err))))))
         ;; File links: support file:, file+*, and raw absolute paths
         (t
          (let* ((abs
                  (cond
                   ((string-match "^file\\(?:\\+[^:]+\\)?:\\(.+\\)$" raw)
                    (match-string 1 raw))
                   ((string-prefix-p "/" raw) raw)
                   (t nil))))
            (when abs
              ;; Strip ::search, unescape, expand, absolutize
              (setq abs (car (split-string abs "::")))
              (when (fboundp 'org-link-unescape)
                (setq abs (org-link-unescape abs)))
              (setq abs (expand-file-name abs (file-name-directory base-file)))
              (let ((canon (ignore-errors (file-truename abs))))
                (when (and canon (file-readable-p canon)
                           (not (member canon file-seen)))
                  (condition-case nil
                      (let ((f-content (with-temp-buffer
                                         (insert-file-contents canon)
                                         (buffer-string))))
                        (let ((file-data (list :id canon
                                              :title desc
                                              :content f-content
                                              :file canon)))
                          (push (list (intern desc) file-data) results))
                        (push canon file-seen))
                    (error nil))))))))))
    (nreverse results)))

(defun my/gptel-org-roam-node-content-by-id (id &optional seen file-seen)
  "Returns memory content with title-keyed structure separating original from required reading."
  (interactive)
  (let* ((seen (or seen (list (concat "id:" id))))
         (file-seen (or file-seen '()))
         (original-memory (my/gptel-get-single-memory-content id)))
    (when original-memory
      (let* ((original-title (plist-get original-memory :title))
             (original-content (plist-get original-memory :content))
             (original-file (plist-get original-memory :file))
             (required-reading-links (my/gptel-extract-required-reading-links 
                                      original-content original-file))
             (required-reading-context (my/gptel-process-required-reading-links 
                                        required-reading-links seen file-seen original-file)))
        ;; Return structure with original memory title as key
        (list (intern original-title) original-memory
              :required-reading-context required-reading-context)))))

  (defun my/org-roam-add-backlink (src-id tgt-id &optional link-text)
    "In SRC-ID node, append a backlink to TGT-ID. Optional LINK-TEXT."
    (let* ((src-node (org-roam-node-from-id src-id))
	   (tgt-node (org-roam-node-from-id tgt-id)))
      (unless (and src-node tgt-node)
	(error "Source or target node not found"))
      (with-current-buffer (find-file-noselect (org-roam-node-file src-node))
	(save-excursion
	  (goto-char (org-roam-node-point src-node))
	  (org-end-of-subtree t)
	  (forward-line)
	  (let ((link (format "[[id:%s][%s]]"
			      tgt-id
			      (or link-text (org-roam-node-title tgt-node)))))
	    (insert link "\n")))
	(save-buffer))
      (message "Backlink added from %s to %s"
	       (org-roam-node-title src-node)
	       (org-roam-node-title tgt-node))))
