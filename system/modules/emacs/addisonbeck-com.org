* Running Axum Server

#+begin_src emacs-lisp
(defcustom addisonbeck-com-serve-command "nix develop --command cargo run --bin addisonbeck-com"
  "Command to run the addisonbeck.com Axum server in development mode"
  :type 'string
  :group 'tools)

(defun addisonbeck-com/serve ()
  "Select the addisonbeck.com project and run the Axum server."
  (interactive)
  (my/projectile-run-command-in-selected-project addisonbeck-com-serve-command))

(add-to-list 'my/executer-picker-candidates
             '("addisonbeck.com: serve axum server" . addisonbeck-com/serve))
#+end_src
