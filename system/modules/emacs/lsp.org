;; Custom functions for extracting LSP data for AI/LLM consumption
(defun lsp-get-buffer-diagnostics-for-ai ()
  "Extract all diagnostics from current buffer in structured format for LLM."
  (interactive)
  (let ((diagnostics (lsp--get-buffer-diagnostics)))
    (mapcar (lambda (diag)
              (let-alist diag
                `((message . ,.message)
                  (severity . ,(pcase .severity
                                (1 "error")
                                (2 "warning") 
                                (3 "info")
                                (4 "hint")))
                  (line . ,(1+ (lsp-translate-line .range.start.line)))
                  (character . ,.range.start.character)
                  (source . ,.source))))
            diagnostics)))
(defun lsp-diagnostics-to-json ()
  "Get current buffer diagnostics as JSON string for LLM consumption."
  (interactive)
  (let ((diagnostics (lsp-get-buffer-diagnostics-for-ai)))
    (json-encode diagnostics)))
(defun lsp-copy-diagnostics-for-ai ()
  "Copy diagnostics to clipboard in LLM-friendly format."
  (interactive)
  (let ((diagnostics (lsp-get-buffer-diagnostics-for-ai)))
    (kill-new (json-encode diagnostics))
    (message "Diagnostics copied to clipboard as JSON")))
